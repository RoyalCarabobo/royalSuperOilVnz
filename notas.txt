CREATE OR REPLACE FUNCTION public.distribuir_nuevo_usuario()
RETURNS TRIGGER AS $$
BEGIN
    -- 1. Si el metadato indica que es un 'cliente'
    IF (new.raw_user_meta_data->>'rol') = 'cliente' THEN
        INSERT INTO public.clientes (user_id, email, nombre_encargado, status, vendedor_id)
        VALUES (
            new.id, 
            new.email, 
            new.raw_user_meta_data->>'nombre_completo', 
            'pendiente',
            (new.raw_user_meta_data->>'vendedor_id')::uuid
        );
    
    -- 2. Si es personal de la empresa (vendedor, admin, etc.)
    ELSE
        INSERT INTO public.usuarios (id, nombre_completo, rol, status)
        VALUES (
            new.id, 
            new.raw_user_meta_data->>'nombre_completo', 
            COALESCE(new.raw_user_meta_data->>'rol', 'vendedor'), 
            'activo'
        );
    END IF;

    RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

CREATE OR REPLACE FUNCTION public.handle_new_user_distribution()
RETURNS TRIGGER AS $$
DECLARE
  v_vendedor_id UUID;
BEGIN
  -- 1. CASO: CLIENTE
  IF (NEW.raw_user_meta_data->>'rol') = 'cliente' THEN
    
    -- Intentamos buscar el vendedor de la zona
    SELECT vendedor_id INTO v_vendedor_id 
    FROM public.zonas_ventas 
    WHERE UPPER(nombre) = UPPER(NEW.raw_user_meta_data->>'zona')
    LIMIT 1;

    -- RESPALDO: Si no hay vendedor en esa zona, asignamos el primero que encontremos
    -- Esto evita que el registro falle por un valor nulo si la columna es obligatoria
    IF v_vendedor_id IS NULL THEN
      SELECT id INTO v_vendedor_id FROM public.usuarios WHERE rol = 'vendedor' LIMIT 1;
    END IF;

    -- INSERCIÓN EN TABLA CLIENTES
    INSERT INTO public.clientes (
      id,
      razon_social,
      rif,
      encargado,
      vendedor_id,
      direccion,
      telefono,
      email,
      zona,
      status,
      fecha_registro
    )
    VALUES (
      NEW.id,
      COALESCE(NEW.raw_user_meta_data->>'razon_social', 'POR DEFINIR'),
      COALESCE(NEW.raw_user_meta_data->>'rif', 'SIN RIF'),
      COALESCE(NEW.raw_user_meta_data->>'nombre_completo', 'SIN NOMBRE'),
      v_vendedor_id, 
      COALESCE(NEW.raw_user_meta_data->>'direccion', 'DIRECCIÓN PENDIENTE'),
      COALESCE(NEW.raw_user_meta_data->>'telefono', 'SIN TELÉFONO'),
      NEW.email,
      COALESCE(NEW.raw_user_meta_data->>'zona', 'GENERAL'),
      'pendiente',
      NOW()
    );

  -- 2. CASO: PERSONAL INTERNO (ADMIN/VENDEDOR)
  ELSE
    INSERT INTO public.usuarios (id, nombre_completo, correo, rol, status)
    VALUES (
      NEW.id,
      COALESCE(NEW.raw_user_meta_data->>'nombre_completo', 'Nuevo Usuario'),
      NEW.email,
      COALESCE(NEW.raw_user_meta_data->>'rol', 'vendedor'),
      'activo'
    );
  END IF;

  RETURN NEW;
EXCEPTION WHEN OTHERS THEN
  -- Este bloque es vital: si algo falla, registra el error en los logs pero no bloquea el registro de Auth
  -- Aunque lo ideal es que el insert funcione, esto te permite ver qué falló sin romper el app
  RAISE LOG 'Error en handle_new_user_distribution: %', SQLERRM;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;
